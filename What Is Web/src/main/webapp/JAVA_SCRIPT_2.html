          prompt() 함수 : String prompt([String message], [String defaultValue])
          문자열을 입력할 때 사용
          첫 번째 매개변수는 입력 창에 출력되는 메시지

          자바스크립트 주석 : // 한줄주석, / * ~ * / 여러줄 주석
          자바스크립트 형변환 :
          var 변수 = parseInt(문자); 문자를 정수형로 변환
          var 변수 = parseFloat(문자); 문자를 실수형으로 변환
          var 변수 = Nember(문자); 문자를 정수&실수형으로 변환
          var 변수 = String(숫자); 숫자를 문자로 변환
          var 변수 = 숫자.toString(진법); 숫자를 문자로 변환
          var 변수 = 숫자.toFixed(소수자리수); 숫자를 문자로 변환, 실수형의 소수점 자리를 지정가능
          var 변수 = 숫자 + "문자열"; 숫자와 문자열을 한 문자열로 더해줌

            var x = document.getElementById("demo");

            function getLocation (){
              if (navigator.geolocation){
                  navigator.geolocation.getCurrentPosition(showPosition);
                  } 
              else{
                    x.innerHTML = "Geolocation is not supported by this browser.";
                  }
            }
            function showPosition (position){
              x.innerHTML = "Latitude: " + position.coords.latitude + "<br>Longitude: " + position.coords.longitude;
            }
